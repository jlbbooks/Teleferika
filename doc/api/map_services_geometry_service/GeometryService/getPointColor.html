Color getPointColor(PointModel point, List<PointModel> allPoints) {
  final angleDeg = calculateAngleAtPoint(point, allPoints);
  if (angleDeg == null) {
    return AppConfig.angleColorGood; // First or last point
  }
  return angleColor(angleDeg);
}
