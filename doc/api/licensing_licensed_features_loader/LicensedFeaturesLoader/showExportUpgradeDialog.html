static void showExportUpgradeDialog(BuildContext context) {
  final l10n = LfpLocalizations.of(context);
  showDialog(
    context: context,
    builder: (BuildContext context) {
      return AlertDialog(
        title: Row(
          children: [
            const Icon(Icons.file_download, color: Colors.blue),
            const SizedBox(width: 8),
            Text(l10n?.exportFeatureTitle ?? 'Export Features'),
          ],
        ),
        content: Column(
          mainAxisSize: MainAxisSize.min,
          crossAxisAlignment: CrossAxisAlignment.start,
          children: [
            Text(
              l10n?.exportFeatureDescription ??
                  'Export functionality is available in this build!',
            ),
            const SizedBox(height: 8),
            Text(l10n?.availableExportFormats ?? 'Available export formats:'),
            Text('• ${l10n?.exportFormatKML ?? 'KML (Google Earth)'}'),
            Text('• ${l10n?.exportFormatCSV ?? 'CSV (Spreadsheets)'}'),
            const SizedBox(height: 8),
            Text(l10n?.featuresIncludeTitle ?? 'Features include:'),
            Text(
              '• ${l10n?.multipleFormatSupportOption ?? 'Multiple format support'}',
            ),
            Text('• ${l10n?.customFileNamingOption ?? 'Custom file naming'}'),
            Text('• ${l10n?.fileSharingOption ?? 'File sharing'}'),
            Text('• ${l10n?.saveToDeviceOption ?? 'Save to device'}'),
          ],
        ),
        actions: [
          TextButton(
            child: Text(l10n?.exportDialogClose ?? 'Close'),
            onPressed: () => Navigator.of(context).pop(),
          ),
        ],
      );
    },
  );
}
