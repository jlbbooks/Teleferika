static List<String> getExportFormats() {
  try {
    final result = FeatureRegistry.executeFunction(
      _pluginName,
      'get_export_formats',
    );
    return (result as List<dynamic>?)?.cast<String>() ?? [];
  } catch (e) {
    logger.severe('Error getting export formats', e);
    return [];
  }
}
