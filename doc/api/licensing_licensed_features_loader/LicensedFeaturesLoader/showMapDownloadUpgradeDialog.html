static void showMapDownloadUpgradeDialog(BuildContext context) {
  final l10n = LfpLocalizations.of(context);
  showDialog(
    context: context,
    builder: (BuildContext context) {
      return AlertDialog(
        title: Row(
          children: [
            const Icon(Icons.map, color: Colors.green),
            const SizedBox(width: 8),
            Text(l10n?.mapDownloadFeatureTitle ?? 'Map Download Features'),
          ],
        ),
        content: Column(
          mainAxisSize: MainAxisSize.min,
          crossAxisAlignment: CrossAxisAlignment.start,
          children: [
            Text(
              l10n?.mapDownloadFeatureDescription ??
                  'Map download functionality is available in this build!',
            ),
            const SizedBox(height: 8),
            Text(l10n?.availableMapTypes ?? 'Available map types:'),
            Text('• ${MapType.of('openStreetMap').name}'),
            Text('• ${MapType.of('satellite').name}'),
            Text('• ${MapType.of('terrain').name}'),
            Text('• ${MapType.of('openTopoMap').name}'),
            Text('• ${MapType.of('thunderforestOutdoors').name}'),
            Text('• ${MapType.of('cartoPositron').name}'),
            Text('• ${MapType.of('thunderforestLandscape').name}'),

            const SizedBox(height: 8),
            Text(
              l10n?.mapDownloadFeaturesIncludeTitle ?? 'Features include:',
            ),
            Text('• ${l10n?.offlineMapAccessOption ?? 'Offline map access'}'),
            Text(
              '• ${l10n?.customAreaSelectionOption ?? 'Custom area selection'}',
            ),
            Text(
              '• ${l10n?.multipleZoomLevelsOption ?? 'Multiple zoom levels'}',
            ),
            Text(
              '• ${l10n?.backgroundDownloadOption ?? 'Background download'}',
            ),
          ],
        ),
        actions: [
          TextButton(
            child: Text(l10n?.mapDownloadDialogClose ?? 'Close'),
            onPressed: () => Navigator.of(context).pop(),
          ),
        ],
      );
    },
  );
}
